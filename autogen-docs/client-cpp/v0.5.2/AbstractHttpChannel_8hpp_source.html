00136 void AbstractHttpChannel<Type>::sync(TransportType type)
00137 {
00138     const auto& supportedTypes = getSupportedTransportTypes();
00139     auto it = supportedTypes.find(type);
00140     if (it != supportedTypes.end() && (it->second == ChannelDirection::UP || it->second == ChannelDirection::BIDIRECTIONAL)) {
00141         KAA_MUTEX_LOCKING("channelGuard_");
00142         boost::unique_lock<boost::mutex> lock(channelGuard_);
00143         KAA_MUTEX_LOCKED("channelGuard_");
00144         if (currentServer_) {
00145             processTypes(std::map<TransportType, ChannelDirection>({ { type, it->second } }), lock);
00146         } else {
00147             lastConnectionFailed_ = true;
00148             KAA_LOG_WARN(boost::format("Can't sync channel %1%. Server is null") % getId());
00149         }
00150     } else {
00151         KAA_LOG_ERROR(boost::format("Unsupported transport type for channel %1%") % getId());
